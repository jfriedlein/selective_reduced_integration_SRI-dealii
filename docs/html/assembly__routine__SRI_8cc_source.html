95  FullMatrix<double> cell_matrix(dofs_per_cell,dofs_per_cell);
96  Vector<double> cell_rhs (dofs_per_cell);
97  // Vector with the indices (global) of the local dofs
98  std::vector<types::global_dof_index> local_dof_indices (dofs_per_cell);
99  // Compute the current, total solution, i.e. starting value of

109  typename DoFHandler<dim>::active_cell_iterator cell = dof_handler_ref.begin_active(),
110  endc = dof_handler_ref.end();

112  for(;cell!=endc;++cell)
113  {
114  // Reset the local rhs and matrix for every cell
115  cell_matrix=0.0;
116  cell_rhs=0.0;
117  // Reinit the FEValues instance for the current cell, i.e.
118  // compute the values for the current cell
119  fe_values_ref.reinit(cell);

121  // [SRI] Also reinit the RI rule for SRI
122  if ( SRI_active )
123  fe_values_ref_RI.reinit(cell);

125  // Vector to store the gradients of the solution at
126  // n_q_points quadrature points
127  std::vector<Tensor<2,dim> > solution_grads_u(n_q_points);
128  // Fill the previous vector using get_function_gradients
129  fe_values_ref[u_fe].get_function_gradients(current_solution,solution_grads_u);
130 
131  // [SRI] Prepare the solutions gradients for the RI rule. Herein we extend
132  // the given vector of solution gradients by the additional gradients at the reduced
133  // integration quadrature points.
134  if ( SRI_active )
135  SRI::prepare_solGrads( n_q_points_RI, fe_values_ref_RI, current_solution, solution_grads_u );

137  // [SRI] k_rel is the relative qp-counter used for everything related to
138  // FEValues objects and needed for SRI
139  unsigned int k_rel = 0;
140 
141  // Write the global indices of the local dofs of the current cell
142  cell->get_dof_indices(local_dof_indices);
143 
144  // Get the QPH for the QPs of the current cell, all stored in one vector of pointers
145  const std::vector< std::shared_ptr< PointHistory<dim> > > lqph = quadrature_point_history.get_data(cell);
146 
147  // [SRI] Loop over all quadrature points of the cell \n
148  // For SRI the variable \a n_q_points_RI contains the number of QPs for reduced integration
149  // over which we also loop. Else (FuI, RI, Fbar) this number is zero, so we don't loop
150  // over these additional QPs
151  for( unsigned int k=0; k < (n_q_points+n_q_points_RI); ++k )
152  {
153  // Compute the deformation gradient from the solution gradients. Note that the
154  // vector of solutions gradients has been, in case of SRI, been extended by the
155  // values at the RI QPs, so we have to use the full QP counter \a k.
156  // @note If you want to work with 2D or even axisymmetry, you would need to
157  // modify the deformation gradient now and cross some more t's (see https://github.com/jfriedlein/2D_axial-symmetry_plane-strain_dealii).
158  //
159  Tensor<2,dim> DeformationGradient = (Tensor<2, dim>(StandardTensors::I<dim>()) + solution_grads_u[k]);
160 
161  // [SRI] SRI (does no harm to the remaining ELFORMs) \n
162  // We initalise the \a fe_values_part pointer with the currently needed
163  // FEValues quantity. For the first \a n_q_points quadrature points we
164  // assign the standard \a fe_values_ref and in the remaining \a n_q_points_RI
165  // we use the \a fe_values_ref_RI for the reduced integration
166  FEValues<dim> *fe_values_part = nullptr;
167  SRI::init_fe_k( /*input->*/ fe_values_ref, fe_values_ref_RI, k, n_q_points,
168  /*output->*/ fe_values_part, k_rel );
169  // [SRI] We declare *_part variables for the stress and the tangent, that
170  // eiter contain the deviatoric or volumetric parts
171  SymmetricTensor<2,dim> stress_part;
172  SymmetricTensor<4,dim> Tangent_part;
173 
174  // The material model will still return the full stress and tangents, because
175  // we call it with the standard deformation gradient. Thus, we also create the
176  // full stress and tangent as tensor. (There is certainly a more efficient way to do this.)
177  SymmetricTensor<2,dim> stress_S;
178  SymmetricTensor<4,dim> dS_dC;
179 
180  // Now you have to call your material model with the deformation gradient and
181  // whatever else, to get your stress and tangent. The following is just a dummy
182  // and far from my actual code.
183  elastoplasticity( DeformationGradient, lqph[k], stress_S, dS_dC );
184 
185  // [SRI] Extract the desired parts from the stress and tangent, in case we use SRI.
186  // Depending on the \a SRI_type, we now do either a volumetric-deviatoric split or
187  // a trivial normal-shear split.
188  if ( SRI_active )
189  {
190  stress_part = SRI::part<dim>( DeformationGradient, stress_S, SRi_type, k, n_q_points );
191  Tangent_part = SRI::part<dim>( DeformationGradient, stress_S, dS_dC, SRI_type, k, n_q_points );
192  }
193  // For full integration, we just copy the full tensors into the *_part variables
194  else
195  {
196  stress_part = stress_S;
197  Tangent_part = dS_dC;
198  }
199 
200  // [SRI] The quadrature weight for the current quadrature point.
201  // That seems to be a daily task, but due to the variable FEValues
202  // element you have to use the general *_part variable and the relative
203  // QP counter \a k_rel
204  const double JxW = (*fe_values_part).JxW(k_rel);
205 
206  // Loop over all dof's of the cell
207  for(unsigned int i=0; i < dofs_per_cell; ++i)
208  {
209  // [SRI] Assemble system_rhs contribution
210  // Here we also access the FEValues, so we replace this by the
211  // *_part counterpart together with \a k-rel
212  Tensor<2,dim> grad_X_N_u_i = (*fe_values_part)[u_fe].gradient(i,k_rel);
213 
214  // [SRI] The standard residual as shown on GitHub. Note that this is exactly the same
215  // as without SRI, only that we replace the PK2 stress by the stress part \stress_part
216  // that either contains the volumetric or deviatoric stress (or normal-shear)
217  cell_rhs(i) -= ( symmetrize( transpose(DeformationGradient) * grad_X_N_u_i ) * stress_part ) * JxW;
218 
219  for(unsigned int j=0; j<dofs_per_cell; ++j)
220  {
221  // [SRI] Assemble tangent contribution
222  Tensor<2,dim> grad_X_N_u_j = (*fe_values_part)[u_fe].gradient(j,k_rel);
223 
224  // The linearisation of the right Cauchy-Green tensor (You will recall this line
225  // when you take a closer look at the F-bar formulation)
226  SymmetricTensor<2,dim> deltaRCG = 2. * symmetrize( transpose(grad_X_N_u_j) * DeformationGradient );
227 
228  // Again, the only difference to the standard integration, is that we replace
229  // the stress and now also the tangent by the *_part counterparts. That's it!
230  cell_matrix(i,j) += (
231  symmetrize( transpose(grad_X_N_u_i) * grad_X_N_u_j ) * stress_part
232  +
233  symmetrize( transpose(DeformationGradient) * grad_X_N_u_i )
234  * ( Tangent_part * deltaRCG )
235  )
236  * JxW;
237  } // end for(j)
238  } // end for(i)
239  } // end for(k)

243  local_dof_indices,
244  tangent_matrix,system_rhs,false);
245  } // end for(cell)
